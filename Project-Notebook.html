final_weather_data = read.csv('final_weather_data.csv')
filtered_data = final_weather_data %>% drop_na()

updated_weather_data = read.csv('new_weather_data.csv')
filtered_weather_data = updated_weather_data %>% drop_na() %>% filter(percent_error_of_forecast != 'Inf')
# day of year vs abs error
ggplot(filtered_data, aes(x=day_of_year, y=absolute_error_of_forecast)) + geom_point() + geom_smooth() + xlab('Day of the Year') + ylab('Real Error of Forecast (F)') + ggtitle('Day of the Year vs Real Error of Forecasted Temperatures')

# observed temperature vs abs error
ggplot(filtered_weather_data, aes(x=observed_temp, y=percent_error_of_forecast)) + geom_point() + geom_smooth() + xlab('Observed Temperature (F)') + ylab('Percent Error of Forecast (%)') + ggtitle('Observed Temperature vs Percent Error of Forecasted Temperatures')
